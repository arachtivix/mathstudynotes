% Define Clojure language for listings
\lstdefinelanguage{Clojure}{
  morekeywords={defn, def, ns, if, do, let, loop, recur, when, cons, car, cdr,

                require, use, in-ns, gen-class},
  morestring=[b]",
  morecomment=[l];,
  sensitive=true,
  alsoletter={\#\-},
  basicstyle=\ttfamily\small,
  keywordstyle=\color{blue}\bfseries,
  stringstyle=\color{red},
  commentstyle=\color{gray}\itshape,
  showstringspaces=false,
  tabsize=2,
  breaklines=true,
  frame=single,
}
